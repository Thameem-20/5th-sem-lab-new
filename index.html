    {
    int i, j, pivot;
    pivot=a[low];
    i=low+1;

    }

    {

    a[low]=a[j];
    a[j]=temp;
    return j;
    }
    }

    public static void Nextvertex(int G[][],int n,int k)
    {

    return;

    }
    }
